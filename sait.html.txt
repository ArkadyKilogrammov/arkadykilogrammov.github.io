<!DOCTYPE html>
<html>
<head>
    <title>Sound Auto-Play</title>
    <style>
        body { 
            margin: 0;
            background: transparent;
        }
    </style>
</head>
<body>
    <audio id="player" autoplay>
        <source src="ne_nado.mp3" type="audio/mpeg">
    </audio>

    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const audio = document.getElementById('player');
            
            // Попытка автовоспроизведения
            audio.play().catch(() => {
                // Создаем кнопку для ручного запуска
                const btn = document.createElement('button');
                btn.textContent = 'Play Sound';
                btn.style = `position:fixed;top:50%;left:50%;transform:translate(-50%,-50%)`;
                btn.onclick = () => audio.play();
                document.body.appendChild(btn);
            });
        });
    </script>
</body>
</html>
